    pass params

    package main

import "fmt"

//! pass params as values vs pass params as references

// I. Basic type: int, float32, string, bool ... => pass by value
func Increase(count int) int {
    count++
    return count
}

// II. Composite type: array, slice, map, struct
// slice, map => pass by ref
// array, struct => pass by value
func UpdateArray(arr [3]int) {
    arr[0] *= 10
    arr[1] *= 10
    arr[2] *= 10
}

func UpdateSlice(slice []int) {
    for i := 0; i < len(slice); i++ {
        slice[i] *= 10
    }
}

func UpdateMap(dict map[int]string) {
    for key, _ := range dict {
        dict[key] = "ABC"
    }
}

type Person struct {
    Name string
    Age  int
}

func UpdatePerson(p Person) {
    p.Name = "Bob"
    p.Age = 20
}

// III. Special type: pointer, interface, variadic params, empty interface
// i) pointer:
func IncreaseWithPointer(count *int) int {
    *count++
    return *count
}
func UpdateArrayWithPointer(arr *[3]int) {
    arr[0] *= 10
    arr[1] *= 10
    arr[2] *= 10
}
func UpdatePersonWithPointer(p *Person) {
    p.Name = "Bob"
    p.Age = 20
}

// ii) interface: ------------------------------------------------------------------------------------------------
type IAccount interface {
    UpdateAccountDetails(name string, balance float64)
    PrintAccountDetails()
}

type Account struct {
    Name    string
    Balance float64
}

type SavingsAccount struct {
    Account      // Embedded type Account
    InterestRate float64
}

type CheckingAccount struct {
    Account        // Embedded type Account
    OverdraftLimit float64
}

type BusinessAccount struct {
    Account     // Embedded type Account
    CompanyName string
}

func (a Account) UpdateAccountDetails(name string, balance float64) {
    a.Name = name
    a.Balance = balance
}

func (a *Account) PrintAccountDetails() {
    fmt.Printf("Account Holder: %s, Balance: %.2f\n", a.Name, a.Balance)
}

func (sa *SavingsAccount) PrintAccountDetails() {
    fmt.Printf("Savings Account Holder: %s, Balance: %.2f, Interest Rate: %.2f%%\n", sa.Name, sa.Balance, sa.InterestRate)
}

func (ca *CheckingAccount) PrintAccountDetails() {
    fmt.Printf("Checking Account Holder: %s, Balance: %.2f, Overdraft Limit: %.2f\n", ca.Name, ca.Balance, ca.OverdraftLimit)
}

func (ba *BusinessAccount) PrintAccountDetails() {
    fmt.Printf("Business Account Holder: %s, Balance: %.2f, Company: %s\n", ba.Name, ba.Balance, ba.CompanyName)
}

func updateAccountDetails(account IAccount, name string, amount float64) {
    account.UpdateAccountDetails(name, amount)
}
// ------------------------------------------------------------------------------------------------

// iii) empty interface:
func modify(any interface{}) { 
    switch v := any.(type) {
    case *int:
        *v *= 2
    }
}

// iv) variadic params:
func mySum(nums ...int) int {
    total := 0
    for idx, num := range nums {
        nums[idx] *= 2
        total += num
    }
    return total
}

func main() {

    // 1. pass by value
    a := 1
    Increase(a)
    fmt.Printf("a = %d \n", a)

    arr := [3]int{1, 2, 3}
    UpdateArray(arr)
    fmt.Printf("UpdateArray: %v \n", arr)

    person := Person{Name: "Viet", Age: 25}
    UpdatePerson(person)
    fmt.Printf("UpdatePerson: %v \n", person)

    // 2. pass by reference (under the hood, it use pointer internal)
    slice := []int{1, 2, 3}
    UpdateSlice(slice)
    fmt.Printf("UpdateSlice: %v \n", slice)

    myMap := map[int]string{
        1: "QWE",
        2: "XYZ",
        3: "UIO",
    }
    UpdateMap(myMap)
    fmt.Printf("UpdateMap: %v \n", myMap)

    // 3. pass by reference (using pointer directly)
    count := 99
    IncreaseWithPointer(&count) // Wow, similar to C++ (-_-)
    fmt.Printf("count = %d \n", count)

    UpdateArrayWithPointer(&arr)
    fmt.Printf("UpdateArrayWithPointer: %v \n", arr)

    UpdatePersonWithPointer(&person)
    fmt.Printf("UpdatePersonWithPointer: %v \n", person)

    // 4. pass a interface (behavior-like pass by reference)
    // the way to achieve polymorphism in Go
    var account IAccount
    account = &SavingsAccount{Account: Account{Name: "Alice", Balance: 1000.0}, InterestRate: 2.5}
    account.PrintAccountDetails()
    updateAccountDetails(account, "Viet111", 11111)
    account.PrintAccountDetails() //? can you guest the output?

    account = &CheckingAccount{Account: Account{Name: "Bob", Balance: 500.0}, OverdraftLimit: 100.0}
    account.PrintAccountDetails()
    updateAccountDetails(account, "Viet222", 22222)
    account.PrintAccountDetails() //? can you guest the output?

    account = &BusinessAccount{Account: Account{Name: "Charlie", Balance: 10000.0}, CompanyName: "Charlie's Chocolates"}
    account.PrintAccountDetails()
    updateAccountDetails(account, "Viet111", 33333)
    account.PrintAccountDetails() //? can you guest the output?

    x := 100
    modify(&x)
    fmt.Printf("x after modify: %d \n", x) //? can you guest the output?

    // 5. pass variadic params
    x1 := 1
    x2 := 2
    x3 := 3
    fmt.Printf("mySum = %d \n", mySum(x1, x2, x3))
    fmt.Printf("x1, x2, x3 = %d %d %d \n", x1, x2, x3) //?  can you guest the output?

}
